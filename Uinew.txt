-- Utility Functions
local function Create(class, props)
    local obj = Instance.new(class)
    for k, v in pairs(props) do
        obj[k] = v
    end
    return obj
end

local function MakeDraggable(frame, dragHandle)
    dragHandle = dragHandle or frame
    local dragging, dragStart, startPos
    dragHandle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)
    dragHandle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement and dragging then
            local delta = input.Position - dragStart
            local cf = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
            TweenService:Create(frame, TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Position = cf}):Play()
        end
    end)
end

local function RippleEffect(button, color)
    color = color or Color3.fromRGB(255, 255, 255)
    button.MouseButton1Click:Connect(function()
        local ripple = Create("Frame", {
            Parent = button,
            BackgroundColor3 = color,
            BackgroundTransparency = 0.6,
            BorderSizePixel = 0,
            Size = UDim2.new(0, 0, 0, 0),
            Position = UDim2.new(0.5, 0, 0.5, 0),
            AnchorPoint = Vector2.new(0.5, 0.5),
            ZIndex = 10
        })
        local baseSize = math.max(button.AbsoluteSize.X, button.AbsoluteSize.Y) * 2
        TweenService:Create(ripple, TweenInfo.new(0.5, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Size = UDim2.new(0, baseSize, 0, baseSize),
            BackgroundTransparency = 1
        }):Play()
        task.delay(0.5, function() ripple:Destroy() end)
    end)
end

-- Settings Manager
local SettingsManager = {}
SettingsManager.__index = SettingsManager

function SettingsManager.new()
    local self = setmetatable({}, SettingsManager)
    self.FileName = "sakura_ui_settings.json"
    self.Settings = {}
    self.Loaded = false
    self:Load()
    return self
end

function SettingsManager:Load()
    local success, data = pcall(function()
        return readfile(self.FileName)
    end)
    if success and data then
        local decoded = game:GetService("HttpService"):JSONDecode(data)
        for k, v in pairs(decoded) do
            if type(v) == "table" and v.R ~= nil then
                decoded[k] = Color3.fromRGB(v.R * 255, v.G * 255, v.B * 255)
            end
        end
        self.Settings = decoded
        self.Loaded = true
    else
        self.Settings = DefaultSettings
        self.Loaded = true
        self:Save()
    end
end

function SettingsManager:Save()
    local data = {}
    for k, v in pairs(self.Settings) do
        if typeof(v) == "Color3" then
            data[k] = {R = v.R, G = v.G, B = v.B}
        else
            data[k] = v
        end
    end
    local success = pcall(function()
        writefile(self.FileName, game:GetService("HttpService"):JSONEncode(data))
    end)
    return success
end

function SettingsManager:Get(key)
    return self.Settings[key]
end

function SettingsManager:Set(key, value)
    self.Settings[key] = value
    self:Save()
end

function SettingsManager:ApplyTheme(themeName)
    local theme = Themes[themeName]
    if not theme then return end
    self.Settings.Primary = theme.Primary
    self.Settings.Background = theme.Background
    self.Settings.Border = theme.Border
    self.Settings.Text = theme.Text
    self.Settings.Icon = theme.Icon
    self.Settings.Theme = themeName
    self:Save()
end

function Module:CalculateContentHeight()
    local h = 0
    for _, control in ipairs(self.Controls) do
        h = h + control.Height
    end
    return h
end

function Module:Toggle(state)
    self.Enabled = state
    
    local contentHeight = self:CalculateContentHeight()
    local targetHeight = self.BaseHeight
    if state then
        targetHeight = 60 + contentHeight + 10
    end
    
    -- Tween module size
    local tweenInfo = TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
    local tween = TweenService:Create(self.Frame, tweenInfo, {Size = UDim2.new(1, -10, 0, targetHeight)})
    tween:Play()
    
    -- Tween icon rotation
    local rotTarget = state and 180 or 0
    TweenService:Create(self.ToggleBtn, TweenInfo.new(0.25), {Rotation = rotTarget}):Play()
    
    -- Fade in/out options
    local opacityTarget = state and 1 or 0
    for _, v in ipairs(self.OptionsContainer:GetChildren()) do
        if v:IsA("GuiObject") then
            TweenService:Create(v, TweenInfo.new(0.2), {BackgroundTransparency = opacityTarget, TextTransparency = opacityTarget}):Play()
        end
    end
    
    -- Notify library to reposition
    self.Library:RepositionModules(self.Frame.Parent)
end

function Module:AddControl(control)
    table.insert(self.Controls, control)
    control.Instance.Parent = self.OptionsContainer
    control.Instance.Visible = true
    
    -- Position control
    local yPos = 0
    for _, c in ipairs(self.Controls) do
        if c == control then break end
        yPos = yPos + c.Height
    end
    control.Instance.Position = UDim2.new(0, 5, 0, yPos)
    
    -- Resize options container
    local totalH = self:CalculateContentHeight()
    self.OptionsContainer.Size = UDim2.new(1, 0, 0, totalH)
    
    -- If enabled, update module height
    if self.Enabled then
        local targetHeight = 60 + totalH + 10
        TweenService:Create(self.Frame, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Size = UDim2.new(1, -10, 0, targetHeight)
        }):Play()
    end
    
    self.Library:RepositionModules(self.Frame.Parent)
end

        BackgroundTransparency = 1,

        TextSize = 12,
        ZIndex = 7
    })

    return paragraph
end

-- Slider
function Module:CreateSlider(data)
    data = data or {}
    local title = data.Title or "Slider"
    local min = data.Min or 0
    local max = data.Max or 100
    local default = data.Default or 50
    local decimals = data.Decimals or 1
    local callback = data.Callback or function() end
    local value = default
    
    local frame = Create("Frame", {
        BackgroundTransparency = 1,
        Size = UDim2.new(1, -10, 0, 45),
        ZIndex = 7
    })
    
    local label = Create("TextLabel", {
        Parent = frame,
        BackgroundTransparency = 1,
        Size = UDim2.new(0.6, 0, 0, 20),
        Position = UDim2.new(0, 5, 0, 0),
        Text = title,
        TextColor3 = self.Library.Settings:Get("Text"),
        TextXAlignment = Enum.TextXAlignment.Left,
        Font = Enum.Font.Gotham,
        TextSize = 14,
        ZIndex = 8
    })
    
    local valueLabel = Create("TextLabel", {
        Parent = frame,
        BackgroundTransparency = 1,
        Size = UDim2.new(0.3, 0, 0, 20),
        Position = UDim2.new(0.7, 0, 0, 0),
        Text = tostring(default),
        TextColor3 = self.Library.Settings:Get("Primary"),
        TextXAlignment = Enum.TextXAlignment.Right,
        Font = Enum.Font.GothamSemibold,
        TextSize = 14,
        ZIndex = 8
    })
    
    -- Slider background
    local sliderBg = Create("Frame", {
        Parent = frame,
        BackgroundColor3 = Color3.fromRGB(200, 200, 200),
        BackgroundTransparency = 0.5,
        Size = UDim2.new(1, -10, 0, 4),
        Position = UDim2.new(0, 5, 0, 28),
        ZIndex = 8
    })
    local sliderCorner = Create("UICorner", {
        Parent = sliderBg,
        CornerRadius = UDim.new(1, 0)
    })
    
    -- Slider fill
    local sliderFill = Create("Frame", {
        Parent = sliderBg,
        BackgroundColor3 = self.Library.Settings:Get("Primary"),
        BorderSizePixel = 0,
        Size = UDim2.new((value - min) / (max - min), 0, 1, 0),
        ZIndex = 9
    })
    local fillCorner = Create("UICorner", {
        Parent = sliderFill,
        CornerRadius = UDim.new(1, 0)
    })
    
    -- Slider knob
    local knob = Create("Frame", {
        Parent = sliderBg,
        BackgroundColor3 = Color3.fromRGB(255, 255, 255),
        Size = UDim2.new(0, 12, 0, 12),
        Position = UDim2.new((value - min) / (max - min), -6, 0.5, -6),
        ZIndex = 10
    })
    local knobCorner = Create("UICorner", {
        Parent = knob,
        CornerRadius = UDim.new(1, 0)
    })
    local knobStroke = Create("UIStroke", {
        Parent = knob,
        Color = self.Library.Settings:Get("Primary"),
        Thickness = 2
    })
    
    local dragging = false
    local function updateSlider(input)
        local pos = input.Position.X - sliderBg.AbsolutePosition.X
        local ratio = math.clamp(pos / sliderBg.AbsoluteSize.X, 0, 1)
        value = min + (max - min) * ratio
        value = math.floor(value * (10 ^ decimals) + 0.5) / (10 ^ decimals)
        value = math.clamp(value, min, max)
        
        sliderFill.Size = UDim2.new(ratio, 0, 1, 0)
        knob.Position = UDim2.new(ratio, -6, 0.5, -6)
        valueLabel.Text = tostring(value)
        
        local success, err = pcall(callback, value)
    end
    
    knob.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
        end
    end)
    
    knob.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)
    
    frame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            updateSlider(input)
            dragging = true
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            updateSlider(input)
        end
    end)
    
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)
    
    self:AddControl({Instance = frame, Height = 48})
    
    return {
        Frame = frame,
        GetValue = function() return value end,
        SetValue = function(v)
            value = math.clamp(v, min, max)
            value = math.floor(value * (10 ^ decimals) + 0.5) / (10 ^ decimals)
            local ratio = (value - min) / (max - min)
            sliderFill.Size = UDim2.new(ratio, 0, 1, 0)
            knob.Position = UDim2.new(ratio, -6, 0.5, -6)
            valueLabel.Text = tostring(value)
        end
    }
end

        CornerRadius = UDim.new(0, 4)

        Color = self.Library.Settings:Get("ModuleBorder"),
        Thickness = 1
    })

    listFrame.ZIndex = 20

            Parent = listFrame,
            BackgroundTransparency = opt == selected and 0.8 or 1,
            Size = UDim2.new(1, 0, 0, 26),

        })

            Parent = optBtn,
            PaddingLeft = UDim.new(0, 8)
        })
        
        optBtn.MouseEnter:Connect(function()
            TweenService:Create(optBtn, TweenInfo.new(0.1), {BackgroundTransparency = 0.85}):Play()
        end)
        optBtn.MouseLeave:Connect(function()

            }):Play()
        end)

            selected = opt
            dropdownBtn.Text = selected

            for _, btn in ipairs(optionBtns) do
                btn.BackgroundTransparency = 1
            end
            optBtn.BackgroundTransparency = 0.8
            
            -- Close dropdown
            toggleDropdown()
            
            local success, err = pcall(callback, selected)
            if not success then warn("Dropdown callback error:", err) end
        end)
        
        table.insert(optionBtns, optBtn)
    end
    
    local function toggleDropdown()
        open = not open
        if open then
            local expandedHeight = math.min(#options * 26, 156)
            listFrame.Size = UDim2.new(0.5, 0, 0, expandedHeight)
            listFrame.Visible = true
            TweenService:Create(listFrame, TweenInfo.new(0.15), {BackgroundTransparency = 0}):Play()

                listFrame.Size = UDim2.new(0.5, 0, 0, 0)
            end)
            arrowLabel.Text = "▾"
        end
    end
    
    dropdownBtn.MouseButton1Click:Connect(toggleDropdown)
    
    -- Close dropdown when clicking outside
    local connection
    connection = UserInputService.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 and open then
            local pos = input.Position
            local absPos = listFrame.AbsolutePosition
            local absSize = listFrame.AbsoluteSize
            if pos.X < absPos.X or pos.X > absPos.X + absSize.X or
               pos.Y < absPos.Y or pos.Y > absPos.Y + absSize.Y then
                if open then toggleDropdown() end
            end
        end
    end)
    
    self:AddControl({Instance = frame, Height = 38})
    
    -- Adjust height when dropdown is open
    return {
        Frame = frame,
        GetValue = function() return selected end,
        SetValue = function(v)
            selected = v
            dropdownBtn.Text = selected
            for _, btn in ipairs(optionBtns) do
                btn.BackgroundTransparency = 1
            end
            for _, btn in ipairs(optionBtns) do
                if btn.Text == selected then
                    btn.BackgroundTransparency = 0.8
                    break
                end
            end
        end
    }
end

    hueKnob.Position = UDim2.new(h, -5, 0.5, -11)
    satKnob.Position = UDim2.new(s, -7, 1 - v, -7)
    satFrame.BackgroundColor3 = Color3.fromHSV(h, 1, 1)
    
    local draggingHue = false
    local draggingSat = false
    
    local function updateFromHSV()
        local color = Color3.fromHSV(h, s, v)
        value = color

            satFrame.BackgroundColor3 = Color3.fromHSV(h, 1, 1)

        end
    end)
    
    satFrame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            draggingSat = true
            local relX = (input.Position.X - satFrame.AbsolutePosition.X) / satFrame.AbsoluteSize.X
            local relY = (input.Position.Y - satFrame.AbsolutePosition.Y) / satFrame.AbsoluteSize.Y
            s = math.clamp(relX, 0, 1)
            v = math.clamp(1 - relY, 0, 1)
            satKnob.Position = UDim2.new(s, -7, 1 - v, -7)
            updateFromHSV()
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement then
            if draggingHue then
                local pos = math.clamp((input.Position.X - hueBar.AbsolutePosition.X) / hueBar.AbsoluteSize.X, 0, 1)
                h = pos
                hueKnob.Position = UDim2.new(pos, -5, 0.5, -11)
                satFrame.BackgroundColor3 = Color3.fromHSV(h, 1, 1)
                updateFromHSV()
            elseif draggingSat then
                local relX = (input.Position.X - satFrame.AbsolutePosition.X) / satFrame.AbsoluteSize.X
                local relY = (input.Position.Y - satFrame.AbsolutePosition.Y) / satFrame.AbsoluteSize.Y
                s = math.clamp(relX, 0, 1)
                v = math.clamp(1 - relY, 0, 1)
                satKnob.Position = UDim2.new(s, -7, 1 - v, -7)
                updateFromHSV()
            end
        end
    end)
    
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            draggingHue = false
            draggingSat = false
        end
    end)
    
    local function togglePicker()
        open = not open
        if open then
            pickerFrame.Visible = true
            TweenService:Create(pickerFrame, TweenInfo.new(0.2), {BackgroundTransparency = 0}):Play()
            frame.Size = UDim2.new(1, -10, 0, 198)
        else
            TweenService:Create(pickerFrame, TweenInfo.new(0.15), {BackgroundTransparency = 0.5}):Play()
            task.delay(0.15, function()
                pickerFrame.Visible = false
                frame.Size = UDim2.new(1, -10, 0, 34)
            end)
        end
    end
    
    preview.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            togglePicker()
        end
    end)
    
    self:AddControl({Instance = frame, Height = 38})
    
    return {
        Frame = frame,
        GetValue = function() return value end,
        SetValue = function(newColor)
            value = newColor
            preview.BackgroundColor3 = newColor
            colorText.Text = string.format("RGB: %d, %d, %d", math.floor(newColor.R * 255), math.floor(newColor.G * 255), math.floor(newColor.B * 255))
            h, s, v = newColor:ToHSV()
            if h then
                hueKnob.Position = UDim2.new(h, -5, 0.5, -11)
                satKnob.Position = UDim2.new(s, -7, 1 - v, -7)
                satFrame.BackgroundColor3 = Color3.fromHSV(h, 1, 1)
            end
        end
    }
}

-- Keybind
function Module:CreateKeybind(data)
    data = data or {}
    local title = data.Title or "Keybind"
    local default = data.Default or Enum.KeyCode.F
    local callback = data.Callback or function() end
    local value = default
    local listening = false
    
    local frame = Create("Frame", {
        BackgroundTransparency = 1,
        Size = UDim2.new(1, -10, 0, 30),
        ZIndex = 7
    })
    
    local label = Create("TextLabel", {
        Parent = frame,
        BackgroundTransparency = 1,
        Size = UDim2.new(0.6, 0, 0, 20),
        Position = UDim2.new(0, 5, 0, 0),
        Text = title,
        TextColor3 = self.Library.Settings:Get("Text"),
        TextXAlignment = Enum.TextXAlignment.Left,
        Font = Enum.Font.Gotham,
        TextSize = 14,
        ZIndex = 8
    })
    
    local keyBtn = Create("TextButton", {
        Parent = frame,
        BackgroundColor3 = self.Library.Settings:Get("Module"),
        BorderSizePixel = 0,
        Size = UDim2.new(0, 70, 0, 24),
        Position = UDim2.new(0.6, 0, 0.5, -12),
        Text = default.Name,
        TextColor3 = self.Library.Settings:Get("Text"),
        Font = Enum.Font.GothamSemibold,
        TextSize = 12,
        ZIndex = 8,
        AutoButtonColor = false
    })
    local keyCorner = Create("UICorner", {
        Parent = keyBtn,
        CornerRadius = UDim.new(0, 4)
    })
    local keyStroke = Create("UIStroke", {
        Parent = keyBtn,
        Color = self.Library.Settings:Get("ModuleBorder"),
        Thickness = 1
    })
    
    keyBtn.MouseButton1Click:Connect(function()
        listening = true
        keyBtn.Text = "..."
        keyBtn.TextColor3 = self.Library.Settings:Get("Primary")
    end)
    
    local connection
    connection = UserInputService.InputBegan:Connect(function(input, processed)
        if processed then return end
        
        if listening then
            if input.KeyCode ~= Enum.KeyCode.Unknown then
                listening = false
                value = input.KeyCode
                keyBtn.Text = value.Name
                keyBtn.TextColor3 = self.Library.Settings:Get("Text")
                local success, err = pcall(callback, value)
            elseif input.UserInputType == Enum.UserInputType.MouseButton1 then
                -- Check if clicking on the button itself
                local absPos = keyBtn.AbsolutePosition
                local absSize = keyBtn.AbsoluteSize
                local pos = input.Position
                if pos.X >= absPos.X and pos.X <= absPos.X + absSize.X and
                   pos.Y >= absPos.Y and pos.Y <= absPos.Y + absSize.Y then
                    -- Continue listening, don't set
                else
                    listening = false
                    keyBtn.Text = value.Name
                    keyBtn.TextColor3 = self.Library.Settings:Get("Text")
                end
            end
        end
    end)
    
    -- Listen for key press to trigger callback
    UserInputService.InputBegan:Connect(function(input, processed)
        if not processed and input.KeyCode == value then
            local success, err = pcall(callback, value)
        end
    end)
    
    self:AddControl({Instance = frame, Height = 34})
    
    return {
        Frame = frame,
        GetValue = function() return value end,
        SetValue = function(kc)
            value = kc
            keyBtn.Text = kc.Name
        end
    }
end

function Tab:Show()
    self.Container.Visible = true
    TweenService:Create(self.Container, TweenInfo.new(0.25, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
        BackgroundTransparency = 0
    }):Play()
end

function Tab:Hide()
    TweenService:Create(self.Container, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
        BackgroundTransparency = 1
    }):Play()
    task.delay(0.2, function()
        self.Container.Visible = false
    end)
end

function Tab:CreateModule(data)
    data = data or {}
    local side = data.Side or "Left"
    local targetContainer = side == "Left" and self.LeftSection or self.RightSection
    
    local module = Module.new(targetContainer, data, self.Library)
    table.insert(self.Modules[side], module)
    self.ModuleCount[side] = self.ModuleCount[side] + 1
    
    self:RepositionModules(targetContainer, side)
    
    return module
end

function Tab:RepositModules(container, side)
    -- Calculate total height and reposition
    local y = 5
    local modules = container:GetChildren()
    for _, child in ipairs(modules) do
        if child:IsA("Frame") and child:FindFirstChild("Header") then
            child.Position = UDim2.new(0, 5, 0, y)
            y = y + child.Size.Y.Offset + 5
        end
    end
    
    if side then
        self.ModuleCount[side] = y
    end
    
    -- Update canvas size
    container.CanvasSize = UDim2.new(0, 0, 0, y + 10)
end

function SakuraUI:ToggleMinimize()
    self.Minimized = not self.Minimized
    
    if self.Minimized then
        -- Minimize: shrink to sidebar
        TweenService:Create(self.MainFrame, TweenInfo.new(0.35, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Size = UDim2.new(0, 180, 0, 60),
            Position = UDim2.new(0.5, -90, 0.5, -30)
        }):Play()
        -- Hide content
        TweenService:Create(self.ContentContainer, TweenInfo.new(0.2), {BackgroundTransparency = 1}):Play()
        self.ContentContainer.Visible = false
    else
        -- Maximize: expand to full
        self.ContentContainer.Visible = true
        TweenService:Create(self.ContentContainer, TweenInfo.new(0.2), {BackgroundTransparency = 0}):Play()
        TweenService:Create(self.MainFrame, TweenInfo.new(0.35, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Size = UDim2.new(0, 760, 0, 520),
            Position = UDim2.new(0.5, -380, 0.5, -260)
        }):Play()
    end
end

function SakuraUI:CreateTab(name, icon)
    icon = icon or ""
    
    -- Create tab data
    local tab = Tab.new({Name = name, Icon = icon}, self)
    table.insert(self.Tabs, tab)
    
    -- Create tab button
    local btn = Create("TextButton", {
        Parent = self.TabContainer,
        BackgroundTransparency = 1,
        Size = UDim2.new(1, 0, 0, 36),
        Position = UDim2.new(0, 0, 0, (#self.Tabs - 1) * 36),
        Text = "",
        ZIndex = 5,
        AutoButtonColor = false
    })
    
    -- Tab icon
    local btnIcon = Create("TextLabel", {
        Parent = btn,
        BackgroundTransparency = 1,
        Size = UDim2.new(0, 24, 1, 0),
        Position = UDim2.new(0, 12, 0, 0),
        Text = icon,
        TextColor3 = self.Settings:Get("Icon"),
        Font = Enum.Font.Gotham,
        TextSize = 16,
        ZIndex = 6
    })
    
    -- Tab label
    local btnLabel = Create("TextLabel", {
        Parent = btn,
        BackgroundTransparency = 1,
        Size = UDim2.new(1, -44, 1, 0),
        Position = UDim2.new(0, 40, 0, 0),
        Text = name,
        TextColor3 = self.Settings:Get("Text"),
        TextXAlignment = Enum.TextXAlignment.Left,
        Font = Enum.Font.Gotham,
        TextSize = 14,
        ZIndex = 6
    })
    
    -- Store reference
    self.TabButtons[tab] = {
        Button = btn,
        Icon = btnIcon,
        Label = btnLabel
    }
    
    -- Update canvas size
    local tabCount = #self.Tabs
    self.TabContainer.CanvasSize = UDim2.new(0, 0, 0, 36 * (tabCount + 1)) -- +1 for settings
    
    -- Click handler
    btn.MouseButton1Click:Connect(function()
        self:SelectTab(tab)
    end)
    
    -- Hover
    btn.MouseEnter:Connect(function()
        if self.ActiveTab ~= tab then
            TweenService:Create(btn, TweenInfo.new(0.15), {BackgroundTransparency = 0.85}):Play()
        end
    end)
    btn.MouseLeave:Connect(function()
        if self.ActiveTab ~= tab then
            TweenService:Create(btn, TweenInfo.new(0.15), {BackgroundTransparency = 1}):Play()
        end
    end)
    
    -- Auto-select first tab
    if #self.Tabs == 1 then
        self:SelectTab(tab)
    end
    
    return tab
end

function SakuraUI:SelectTab(tab)
    -- Hide current tab
    if self.ActiveTab then
        self.ActiveTab:Hide()
    end
    
    -- Show new tab
    self.ActiveTab = tab
    tab:Show()
    
    -- Move indicator
    local btnData = self.TabButtons[tab]
    if btnData then
        local targetY = btnData.Button.Position.Y.Offset
        TweenService:Create(self.TabIndicator, TweenInfo.new(0.25, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Position = UDim2.new(0, 2, 0, targetY + 2)
        }):Play()
        
        -- Highlight active tab
        for _, data in pairs(self.TabButtons) do
            data.Button.BackgroundTransparency = (data == btnData) and 0.85 or 1
        end
    end
end

function SakuraUI:UpdateThemeColors()
    -- Update main frame colors
    TweenService:Create(self.MainFrame, TweenInfo.new(0.3), {
        BackgroundColor3 = self.Settings:Get("Background")
    }):Play()
    TweenService:Create(self.MainStroke, TweenInfo.new(0.3), {
        Color = self.Settings:Get("Border")
    }):Play()
    
    -- Update sidebar
    local sidebarColor = self.Settings:Get("Sidebar") or self.Settings:Get("Background")
    TweenService:Create(self.Sidebar, TweenInfo.new(0.3), {
        BackgroundColor3 = sidebarColor
    }):Play()
    TweenService:Create(self.SidebarCover, TweenInfo.new(0.3), {
        BackgroundColor3 = sidebarColor
    }):Play()
    
    -- Update sidebar divider
    TweenService:Create(self.SidebarDivider, TweenInfo.new(0.3), {
        BackgroundColor3 = self.Settings:Get("Border")
    }):Play()
    
    -- Update indicator
    TweenService:Create(self.TabIndicator, TweenInfo.new(0.3), {
        BackgroundColor3 = self.Settings:Get("Primary")
    }):Play()
    
    -- Update logo icon color
    TweenService:Create(self.LogoIcon, TweenInfo.new(0.3), {
        TextColor3 = self.Settings:Get("Primary")
    }):Play()
    
    -- Update title color
    TweenService:Create(self.UITitle, TweenInfo.new(0.3), {
        TextColor3 = self.Settings:Get("Text")
    }):Play()
    
    -- Update minimize button
    TweenService:Create(self.MinimizeBtn, TweenInfo.new(0.3), {
        ImageColor3 = self.Settings:Get("Icon")
    }):Play()
    
    -- Update tab buttons
    for _, data in pairs(self.TabButtons) do
        TweenService:Create(data.Icon, TweenInfo.new(0.3), {
            TextColor3 = self.Settings:Get("Icon")
        }):Play()
        TweenService:Create(data.Label, TweenInfo.new(0.3), {
            TextColor3 = self.Settings:Get("Text")
        }):Play()
    end
    
    -- Update scrollbar colors
    for _, tab in ipairs(self.Tabs) do
        tab.LeftSection.ScrollBarImageColor3 = self.Settings:Get("Primary")
        tab.RightSection.ScrollBarImageColor3 = self.Settings:Get("Primary")
    end
    
    -- Update module backgrounds within tabs
    for _, tab in ipairs(self.Tabs) do
        for _, side in ipairs({"Left", "Right"}) do
            for _, mod in ipairs(tab.Modules[side]) do
                if mod.Frame then
                    local moduleColor = self.Settings:Get("Module") or 
                        (self.Settings:Get("DarkUI") and Color3.fromRGB(40, 32, 40) or Color3.fromRGB(255, 240, 243))
                    local moduleBorder = self.Settings:Get("ModuleBorder") or
                        (self.Settings:Get("DarkUI") and Color3.fromRGB(60, 45, 55) or Color3.fromRGB(255, 200, 210))
                    
                    TweenService:Create(mod.Frame, TweenInfo.new(0.3), {
                        BackgroundColor3 = moduleColor
                    }):Play()
                    if mod.Border then
                        TweenService:Create(mod.Border, TweenInfo.new(0.3), {
                            Color = moduleBorder
                        }):Play()
                    end
                    
                    -- Update child elements
                    if mod.TitleLabel then
                        TweenService:Create(mod.TitleLabel, TweenInfo.new(0.3), {
                            TextColor3 = self.Settings:Get("Text")
                        }):Play()
                    end
                    if mod.DescLabel then
                        TweenService:Create(mod.DescLabel, TweenInfo.new(0.3), {
                            TextColor3 = self.Settings:Get("TextSecondary") or Color3.fromRGB(180, 120, 120)
                        }):Play()
                    end
                    if mod.Divider then
                        TweenService:Create(mod.Divider, TweenInfo.new(0.3), {
                            BackgroundColor3 = self.Settings:Get("Divider")
                        }):Play()
                    end
                    if mod.ToggleBtn then
                        TweenService:Create(mod.ToggleBtn, TweenInfo.new(0.3), {
                            ImageColor3 = self.Settings:Get("Icon")
                        }):Play()
                    end
                end
            end
        end
    end
end

--[[
    Return the library
]]
return SakuraUI
